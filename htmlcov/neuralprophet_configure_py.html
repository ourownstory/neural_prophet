199 def get_reg_delay_weight(self, e, iter_progress, reg_start_pct: float = 0.6666, reg_full_pct: float = 1): 

200 progress = (e + iter_progress) / float(self.epochs) 

201 if reg_start_pct == reg_full_pct: 

202 reg_progress = float(progress > reg_start_pct) 

203 else: 

204 reg_progress = (progress - reg_start_pct) / (reg_full_pct - reg_start_pct) 

205 if reg_progress <= 0: 

206 delay_weight = 0 

207 elif reg_progress < 1: 

208 delay_weight = 1 - (1 + np.cos(np.pi * reg_progress)) / 2.0 

209 else: 

210 delay_weight = 1 

211 return delay_weight
