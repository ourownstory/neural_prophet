79def create_optimizer(optimizer, model_parameters, lr): 

80 if type(optimizer) == str: 

81 if optimizer.lower() == "adamw": 

82 # Tends to overfit, but reliable 

83 optimizer = torch.optim.AdamW(model_parameters, lr=lr, weight_decay=1e-3)
