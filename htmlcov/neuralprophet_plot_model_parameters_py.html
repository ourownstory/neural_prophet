29def plot_parameters(m, forecast_in_focus=None, weekly_start=0, yearly_start=0, figsize=None): 

30 """Plot the parameters that the model is composed of, visually. 

31 

32 Args: 

33 m (NeuralProphet): fitted model. 

34 forecast_in_focus (int): n-th step ahead forecast AR-coefficients to plot 

35 weekly_start (int): specifying the start day of the weekly seasonality plot. 

36 0 (default) starts the week on Sunday. 

37 1 shifts by 1 day to Monday, and so on. 

38 yearly_start (int): specifying the start day of the yearly seasonality plot. 

39 0 (default) starts the year on Jan 1. 

40 1 shifts by 1 day to Jan 2, and so on. 

41 figsize (tuple): width, height in inches. 

42 None (default): automatic (10, 3 * npanel) 

43 

44 Returns: 

45 A matplotlib figure. 

46 """ 

47 # Identify components to be plotted 

48 # as dict: {plot_name, } 

49 components = [{"plot_name": "Trend"}] 

50 if m.config_trend.n_changepoints > 0: 

51 components.append({"plot_name": "Trend Rate Change"}) 

52 

53 # Plot seasonalities, if present 

54 if m.season_config is not None: 

55 for name in m.season_config.periods: 

56 components.append({"plot_name": "seasonality", "comp_name": name}) 

57 

58 if m.n_lags > 0: 

59 components.append( 

60 { 

61 "plot_name": "lagged weights", 

62 "comp_name": "AR", 

63 "weights": m.model.ar_weights.detach().numpy(), 

64 "focus": forecast_in_focus, 

65 } 

66 ) 

67 

68 # all scalar regressors will be plotted together 

69 # collected as tuples (name, weights) 

70 

71 # Add Regressors 

72 additive_future_regressors = [] 

73 multiplicative_future_regressors = [] 

74 if m.regressors_config is not None: 

75 for regressor, configs in m.regressors_config.items(): 

76 mode = configs["mode"] 

77 regressor_param = m.model.get_reg_weights(regressor) 

78 if mode == "additive": 

79 additive_future_regressors.append((regressor, regressor_param.detach().numpy())) 

80 else: 

81 multiplicative_future_regressors.append((regressor, regressor_param.detach().numpy())) 

82 

83 additive_events = [] 

84 multiplicative_events = [] 

85 # Add Events 

86 # add the country holidays 

87 if m.country_holidays_config is not None: 

88 for country_holiday in m.country_holidays_config["holiday_names"]: 

89 event_params = m.model.get_event_weights(country_holiday) 

90 weight_list = [(key, param.detach().numpy()) for key, param in event_params.items()] 

91 mode = m.country_holidays_config["mode"] 

92 if mode == "additive": 

93 additive_events = additive_events + weight_list 

94 else: 

95 multiplicative_events = multiplicative_events + weight_list 

96 

97 # add the user specified events 

98 if m.events_config is not None: 

99 for event, configs in m.events_config.items(): 

100 event_params = m.model.get_event_weights(event) 

101 weight_list = [(key, param.detach().numpy()) for key, param in event_params.items()] 

102 mode = configs["mode"] 

103 if mode == "additive": 

104 additive_events = additive_events + weight_list 

105 else: 

106 multiplicative_events = multiplicative_events + weight_list 

107 

108 # Add Covariates 

109 lagged_scalar_regressors = [] 

110 if m.config_covar is not None: 

111 for name in m.config_covar.keys(): 

112 if m.config_covar[name].as_scalar: 

113 lagged_scalar_regressors.append((name, m.model.get_covar_weights(name).detach().numpy())) 

114 else: 

115 components.append( 

116 { 

117 "plot_name": "lagged weights", 

118 "comp_name": 'Lagged Regressor "{}"'.format(name), 

119 "weights": m.model.get_covar_weights(name).detach().numpy(), 

120 "focus": forecast_in_focus, 

121 } 

122 ) 

123 

124 if len(additive_future_regressors) > 0: 

125 components.append({"plot_name": "Additive future regressor"}) 

126 if len(multiplicative_future_regressors) > 0: 

127 components.append({"plot_name": "Multiplicative future regressor"}) 

128 if len(lagged_scalar_regressors) > 0: 

129 components.append({"plot_name": "Lagged scalar regressor"}) 

130 if len(additive_events) > 0: 

131 additive_events = [(key, weight * m.data_params["y"].scale) for (key, weight) in additive_events] 

132 

133 components.append({"plot_name": "Additive event"}) 

134 if len(multiplicative_events) > 0: 

135 components.append({"plot_name": "Multiplicative event"}) 

136 

137 npanel = len(components) 

138 figsize = figsize if figsize else (10, 3 * npanel) 

139 fig, axes = plt.subplots(npanel, 1, facecolor="w", figsize=figsize) 

140 if npanel == 1: 

141 axes = [axes] 

142 multiplicative_axes = [] 

143 for ax, comp in zip(axes, components): 

144 plot_name = comp["plot_name"].lower() 

145 if plot_name.startswith("trend"): 

146 if "change" in plot_name: 

147 plot_trend_change(m=m, ax=ax, plot_name=comp["plot_name"]) 

148 else: 

149 plot_trend(m=m, ax=ax, plot_name=comp["plot_name"]) 

150 elif plot_name.startswith("seasonality"): 

151 name = comp["comp_name"] 

152 if m.season_config.mode == "multiplicative": 

153 multiplicative_axes.append(ax) 

154 if name.lower() == "weekly" or m.season_config.periods[name].period == 7: 

155 plot_weekly(m=m, ax=ax, weekly_start=weekly_start, comp_name=name) 

156 elif name.lower() == "yearly" or m.season_config.periods[name].period == 365.25: 

157 plot_yearly(m=m, ax=ax, yearly_start=yearly_start, comp_name=name) 

158 elif name.lower() == "daily" or m.season_config.periods[name].period == 1: 

159 plot_daily(m=m, ax=ax, comp_name=name) 

160 else: 

161 plot_custom_season(m=m, ax=ax, comp_name=name) 

162 elif plot_name == "lagged weights": 

163 plot_lagged_weights(weights=comp["weights"], comp_name=comp["comp_name"], focus=comp["focus"], ax=ax) 

164 else: 

165 if plot_name == "additive future regressor": 

166 weights = additive_future_regressors 

167 elif plot_name == "multiplicative future regressor": 

168 multiplicative_axes.append(ax) 

169 weights = multiplicative_future_regressors 

170 elif plot_name == "lagged scalar regressor": 

171 weights = lagged_scalar_regressors 

172 elif plot_name == "additive event": 

173 weights = additive_events 

174 elif plot_name == "multiplicative event": 

175 multiplicative_axes.append(ax) 

176 weights = multiplicative_events 

177 plot_scalar_weights(weights=weights, plot_name=comp["plot_name"], focus=forecast_in_focus, ax=ax) 

178 fig.tight_layout() 

179 # Reset multiplicative axes labels after tight_layout adjustment 

180 for ax in multiplicative_axes: 

181 ax = set_y_as_percent(ax) 

182 return fig

185def plot_trend_change(m, ax=None, plot_name="Trend Change", figsize=(10, 6)): 

186 """Make a barplot of the magnitudes of trend-changes. 

187 

188 Args: 

189 m (NeuralProphet): fitted model. 

190 ax (matplotlib axis): matplotlib Axes to plot on. 

191 One will be created if this is not provided. 

192 plot_name (str): Name of the plot Title. 

193 figsize (tuple): width, height in inches. Ignored if ax is not None. 

194 default: (10, 6) 

195 

196 Returns: 

197 a list of matplotlib artists 

198 """ 

199 artists = [] 

200 if not ax: 

201 fig = plt.figure(facecolor="w", figsize=figsize) 

202 ax = fig.add_subplot(111) 

203 

204 start = m.data_params["ds"].shift 

205 scale = m.data_params["ds"].scale 

206 time_span_seconds = scale.total_seconds() 

207 cp_t = [] 

208 for cp in m.model.config_trend.changepoints: 

209 cp_t.append(start + datetime.timedelta(seconds=cp * time_span_seconds)) 

210 weights = m.model.get_trend_deltas.detach().numpy() 

211 # add end-point to force scale to match trend plot 

212 cp_t.append(start + scale) 

213 weights = np.append(weights, [0.0]) 

214 width = time_span_seconds / 175000 / m.config_trend.n_changepoints 

215 artists += ax.bar(cp_t, weights, width=width, color="#0072B2") 

216 locator = AutoDateLocator(interval_multiples=False) 

217 formatter = AutoDateFormatter(locator) 

218 ax.xaxis.set_major_locator(locator) 

219 ax.xaxis.set_major_formatter(formatter) 

220 ax.grid(True, which="major", c="gray", ls="-", lw=1, alpha=0.2) 

221 ax.set_xlabel("Trend Segment") 

222 ax.set_ylabel(plot_name) 

223 return artists

227 """Make a barplot of the magnitudes of trend-changes.

230 m (NeuralProphet): fitted model.

272def plot_scalar_weights(weights, plot_name, focus=None, ax=None, figsize=(10, 6)): 

273 """Make a barplot of the regressor weights. 

274 

275 Args: 

276 weights (list): tuples (name, weights) 

277 plot_name (string): name of the plot 

278 ax (matplotlib axis): matplotlib Axes to plot on. 

279 One will be created if this is not provided. 

280 focus (int): if provided, show weights for this forecast 

281 None (default) plot average 

282 figsize (tuple): width, height in inches. Ignored if ax is not None. 

283 default: (10, 6) 

284 Returns: 

285 a list of matplotlib artists 

286 """ 

287 artists = [] 

288 if not ax: 

289 fig = plt.figure(facecolor="w", figsize=figsize) 

290 ax = fig.add_subplot(111) 

291 # if len(regressors) == 1: 

292 # else: 

293 names = [] 

294 values = [] 

295 for name, weights in weights: 

296 names.append(name) 

297 weight = np.squeeze(weights) 

298 if len(weight.shape) > 1: 

299 raise ValueError("Not scalar " + plot_name) 

300 if len(weight.shape) == 1 and len(weight) > 1: 

301 if focus is not None: 

302 weight = weight[focus - 1] 

303 else: 

304 weight = np.mean(weight) 

305 values.append(weight) 

306 artists += ax.bar(names, values, width=0.8, color="#0072B2") 

307 ax.grid(True, which="major", c="gray", ls="-", lw=1, alpha=0.2) 

308 ax.set_xlabel(plot_name + " name") 

309 xticks = ax.get_xticklabels() 

310 if len("_".join(names)) > 100: 

311 for tick in xticks: 

312 tick.set_ha("right") 

313 tick.set_rotation(20) 

314 if "lagged" in plot_name.lower(): 

315 if focus is None: 

316 ax.set_ylabel(plot_name + " weight (avg)") 

317 else: 

318 ax.set_ylabel(plot_name + " weight ({})-ahead".format(focus)) 

319 else: 

320 ax.set_ylabel(plot_name + " weight") 

321 return artists

324def plot_lagged_weights(weights, comp_name, focus=None, ax=None, figsize=(10, 6)): 

325 """Make a barplot of the importance of lagged inputs. 

326 

327 Args: 

328 weights (np.array): model weights as matrix or vector 

329 comp_name (str): name of lagged inputs 

330 focus (int): if provided, show weights for this forecast 

331 None (default) sum over all forecasts and plot as relative percentage 

332 ax (matplotlib axis): matplotlib Axes to plot on. 

333 One will be created if this is not provided. 

334 figsize (tuple): width, height in inches. Ignored if ax is not None. 

335 default: (10, 6) 

336 Returns: 

337 a list of matplotlib artists 

338 """ 

339 artists = [] 

340 if not ax: 

341 fig = plt.figure(facecolor="w", figsize=figsize) 

342 ax = fig.add_subplot(111) 

343 n_lags = weights.shape[1] 

344 lags_range = list(range(1, 1 + n_lags))[::-1] 

345 if focus is None: 

346 weights = np.sum(np.abs(weights), axis=0) 

347 weights = weights / np.sum(weights) 

348 artists += ax.bar(lags_range, weights, width=1.00, color="#0072B2") 

349 else: 

350 if len(weights.shape) == 2: 

351 weights = weights[focus - 1, :] 

352 artists += ax.bar(lags_range, weights, width=0.80, color="#0072B2") 

353 ax.grid(True, which="major", c="gray", ls="-", lw=1, alpha=0.2) 

354 ax.set_xlabel("{} lag number".format(comp_name)) 

355 if focus is None: 

356 ax.set_ylabel("{} relevance".format(comp_name)) 

357 ax = set_y_as_percent(ax) 

358 else: 

359 ax.set_ylabel("{} weight ({})-ahead".format(comp_name, focus)) 

360 return artists

363def predict_one_season(m, name, n_steps=100): 

364 config = m.season_config.periods[name] 

365 t_i = np.arange(n_steps + 1) / float(n_steps) 

366 features = time_dataset.fourier_series_t( 

367 t=t_i * config.period, period=config.period, series_order=config.resolution 

368 ) 

369 features = torch.from_numpy(np.expand_dims(features, 1)) 

370 predicted = m.model.seasonality(features=features, name=name) 

371 predicted = predicted.squeeze().detach().numpy() 

372 if m.season_config.mode == "additive": 

373 predicted = predicted * m.data_params["y"].scale 

374 return t_i, predicted

379 features = time_dataset.fourier_series(dates=dates, period=config.period, series_order=config.resolution) 

380 features = torch.from_numpy(np.expand_dims(features, 1)) 

381 predicted = m.model.seasonality(features=features, name=name) 

382 predicted = predicted.squeeze().detach().numpy() 

383 if m.season_config.mode == "additive": 

384 predicted = predicted * m.data_params["y"].scale 

385 return predicted

414def plot_yearly(m, comp_name="yearly", yearly_start=0, quick=True, ax=None, figsize=(10, 6)): 

415 """Plot the yearly component of the forecast. 

416 

417 Args: 

418 m (NeuralProphet): fitted model. 

419 ax (matplotlib axis): matplotlib Axes to plot on. 

420 One will be created if this is not provided. 

421 yearly_start (int): specifying the start day of the yearly seasonality plot. 

422 0 (default) starts the year on Jan 1. 

423 1 shifts by 1 day to Jan 2, and so on. 

424 quick (bool): use quick low-evel call of model. might break in future. 

425 figsize (tuple): width, height in inches. Ignored if ax is not None. 

426 default: (10, 6) 

427 comp_name (str): Name of seasonality component if previously changed from default 'yearly'. 

428 

429 Returns: 

430 a list of matplotlib artists 

431 """ 

432 artists = [] 

433 if not ax: 

434 fig = plt.figure(facecolor="w", figsize=figsize) 

435 ax = fig.add_subplot(111) 

436 # Compute yearly seasonality for a Jan 1 - Dec 31 sequence of dates. 

437 days = pd.date_range(start="2017-01-01", periods=365) + pd.Timedelta(days=yearly_start) 

438 df_y = pd.DataFrame({"ds": days}) 

439 if quick: 

440 predicted = predict_season_from_dates(m, dates=df_y["ds"], name=comp_name) 

441 else: 

442 predicted = m.predict_seasonal_components(df_y)[comp_name] 

443 artists += ax.plot(df_y["ds"].dt.to_pydatetime(), predicted, ls="-", c="#0072B2") 

444 ax.grid(True, which="major", c="gray", ls="-", lw=1, alpha=0.2)

446 ax.xaxis.set_major_formatter(FuncFormatter(lambda x, pos=None: "{dt:%B} {dt.day}".format(dt=num2date(x)))) 

447 ax.xaxis.set_major_locator(months) 

448 ax.set_xlabel("Day of year") 

449 ax.set_ylabel("Seasonality: {}".format(comp_name))
