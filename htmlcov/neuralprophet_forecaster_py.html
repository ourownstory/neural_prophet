28 """NeuralProphet forecaster. 

29 

30 A simple yet powerful forecaster that models:

62 """ 

63 Args: 

64 ## Trend Config 

65 growth (str): ['off', 'linear'] to specify 

66 no trend or a linear trend. 

67 Note: 'discontinuous' setting is actually not a trend per se. only use if you know what you do. 

68 changepoints list: Dates at which to include potential changepoints. 

69 If not specified, potential changepoints are selected automatically. 

70 data format: list of str, list of np.datetimes, np.array of np.datetimes (not np.array of np.str) 

71 n_changepoints (int): Number of potential changepoints to include. 

72 Changepoints are selected uniformly from the first `changepoint_range` proportion of the history. 

73 Not used if input `changepoints` is supplied. If `changepoints` is not supplied. 

74 changepoints_range (float): Proportion of history in which trend changepoints will 

75 be estimated. Defaults to 0.8 for the first 80%. Not used if `changepoints` is specified. 

76 trend_reg (float): Parameter modulating the flexibility of the automatic changepoint selection. 

77 Large values (~1-100) will limit the variability of changepoints. 

78 Small values (~0.001-1.0) will allow changepoints to change faster. 

79 default: 0 will fully fit a trend to each segment. 

80 trend_reg_threshold (bool, float): Allowance for trend to change without regularization. 

81 True: Automatically set to a value that leads to a smooth trend. 

82 False: All changes in changepoints are regularized 

83 

84 ## Seasonality Config 

85 yearly_seasonality (bool, int): Fit yearly seasonality. 

86 Can be 'auto', True, False, or a number of Fourier/linear terms to generate. 

87 weekly_seasonality (bool, int): Fit monthly seasonality. 

88 Can be 'auto', True, False, or a number of Fourier/linear terms to generate. 

89 daily_seasonality (bool, int): Fit daily seasonality. 

90 Can be 'auto', True, False, or a number of Fourier/linear terms to generate. 

91 seasonality_mode (str): 'additive' (default) or 'multiplicative'. 

92 seasonality_reg (float): Parameter modulating the strength of the seasonality model. 

93 Smaller values (~0.1-1) allow the model to fit larger seasonal fluctuations, 

94 larger values (~1-100) dampen the seasonality. 

95 default: None, no regularization 

96 

97 ## AR Config 

98 n_lags (int): Previous time series steps to include in auto-regression. Aka AR-order 

99 ar_sparsity (float): [0-1], how much sparsity to enduce in the AR-coefficients. 

100 Should be around (# nonzero components) / (AR order), eg. 3/100 = 0.03 

101 

102 ## Model Config 

103 n_forecasts (int): Number of steps ahead of prediction time step to forecast. 

104 num_hidden_layers (int): number of hidden layer to include in AR-Net. defaults to 0. 

105 d_hidden (int): dimension of hidden layers of the AR-Net. Ignored if num_hidden_layers == 0. 

106 

107 ## Train Config 

108 learning_rate (float): Maximum learning rate setting for 1cycle policy scheduler. 

109 default: None: Automatically sets the learning_rate based on a learning rate range test. 

110 For manual values, try values ~0.001-10. 

111 epochs (int): Number of epochs (complete iterations over dataset) to train model. 

112 default: None: Automatically sets the number of epochs based on dataset size. 

113 For best results also leave batch_size to None. 

114 For manual values, try ~5-500. 

115 batch_size (int): Number of samples per mini-batch. 

116 default: None: Automatically sets the batch_size based on dataset size. 

117 For best results also leave epochs to None. 

118 For manual values, try ~1-512. 

119 loss_func (str, torch.nn.modules.loss._Loss, 'typing.Callable'): 

120 Type of loss to use: str ['Huber', 'MSE'], 

121 or torch loss or callable for custom loss, eg. asymmetric Huber loss 

122 train_speed (int, float) a quick setting to speed up or slow down model fitting [-3, -2, -1, 0, 1, 2, 3] 

123 potentially useful when under-, over-fitting, or simply in a hurry. 

124 applies epochs *= 2**-train_speed, batch_size *= 2**train_speed, learning_rate *= 2**train_speed, 

125 default None: equivalent to 0. 

126 

127 ## Data config 

128 normalize (str): Type of normalization to apply to the time series. 

129 options: ['auto', 'soft', 'off', 'minmax, 'standardize'] 

130 default: 'auto' uses 'minmax' if variable is binary, else 'soft' 

131 'soft' scales minimum to 0.1 and the 90th quantile to 0.9 

132 impute_missing (bool): whether to automatically impute missing dates/values 

133 imputation follows a linear method up to 10 missing values, more are filled with trend. 

134 """

258 def _handle_missing_data(self, df, freq, predicting=False): 

259 """Checks, auto-imputes and normalizes new data 

260 

261 Args: 

262 df (pd.DataFrame): raw data with columns 'ds' and 'y' 

263 freq (str): data frequency 

264 predicting (bool): when no lags, allow NA values in 'y' of forecast series or 'y' to miss completely 

265 

266 Returns: 

267 pre-processed df 

268 """ 

269 if self.n_lags == 0 and not predicting: 

270 # we can drop rows with NA in y 

271 sum_na = sum(df["y"].isna()) 

272 if sum_na > 0: 

273 df = df[df["y"].notna()] 

274 log.info("dropped {} NAN row in 'y'".format(sum_na)) 

275 

276 # add missing dates for autoregression modelling 

277 if self.n_lags > 0: 

278 df, missing_dates = df_utils.add_missing_dates_nan(df, freq=freq) 

279 if missing_dates > 0: 

280 if self.impute_missing: 

281 log.info("{} missing dates added.".format(missing_dates)) 

282 else: 

283 raise ValueError( 

284 "{} missing dates found. Please preprocess data manually or set impute_missing to True.".format( 

285 missing_dates 

286 ) 

287 ) 

288 

289 # impute missing values 

290 data_columns = [] 

291 if self.n_lags > 0: 

292 data_columns.append("y") 

293 if self.config_covar is not None: 

294 data_columns.extend(self.config_covar.keys()) 

295 if self.regressors_config is not None: 

296 data_columns.extend(self.regressors_config.keys()) 

297 if self.events_config is not None: 

298 data_columns.extend(self.events_config.keys()) 

299 for column in data_columns: 

300 sum_na = sum(df[column].isnull()) 

301 if sum_na > 0: 

302 if self.impute_missing: 

303 # use 0 substitution for holidays and events missing values 

304 if self.events_config is not None and column in self.events_config.keys(): 

305 df[column].fillna(0, inplace=True) 

306 remaining_na = 0 

307 else: 

308 df.loc[:, column], remaining_na = df_utils.fill_linear_then_rolling_avg( 

309 df[column], 

310 limit_linear=self.impute_limit_linear, 

311 rolling=self.impute_rolling, 

312 ) 

313 log.info("{} NaN values in column {} were auto-imputed.".format(sum_na - remaining_na, column)) 

314 if remaining_na > 0: 

315 raise ValueError( 

316 "More than {} consecutive missing values encountered in column {}. " 

317 "{} NA remain. Please preprocess data manually.".format( 

318 2 * self.impute_limit_linear + self.impute_rolling, column, remaining_na 

319 ) 

320 ) 

321 else: # fail because set to not impute missing 

322 raise ValueError( 

323 "Missing values found. " "Please preprocess data manually or set impute_missing to True." 

324 ) 

325 return df

440 def _train_epoch(self, e, loader): 

441 """Make one complete iteration over all samples in dataloader and update model after each batch. 

442 

443 Args: 

444 e (int): current epoch number 

445 loader (torch DataLoader): Training Dataloader 

446 """ 

447 self.model.train() 

448 for i, (inputs, targets) in enumerate(loader): 

449 # Run forward calculation 

450 predicted = self.model.forward(inputs) 

451 # Compute loss. 

452 loss = self.config_train.loss_func(predicted, targets) 

453 # Regularize. 

454 loss, reg_loss = self._add_batch_regualarizations(loss, e, i / float(len(loader))) 

455 self.optimizer.zero_grad() 

456 loss.backward() 

457 self.optimizer.step() 

458 self.scheduler.step() 

459 self.metrics.update( 

460 predicted=predicted.detach(), target=targets.detach(), values={"Loss": loss, "RegLoss": reg_loss} 

461 ) 

462 epoch_metrics = self.metrics.compute(save=True) 

463 return epoch_metrics

465 def _add_batch_regualarizations(self, loss, e, iter_progress): 

466 """Add regulatization terms to loss, if applicable 

467 

468 Args: 

469 loss (torch Tensor, scalar): current batch loss 

470 e (int): current epoch number 

471 iter_progress (float): this epoch's progress of iterating over dataset [0, 1] 

472 

473 Returns: 

474 loss, reg_loss 

475 """ 

476 delay_weight = self.config_train.get_reg_delay_weight(e, iter_progress) 

477 

478 reg_loss = torch.zeros(1, dtype=torch.float, requires_grad=False) 

479 if delay_weight > 0: 

480 # Add regularization of AR weights - sparsify 

481 if self.model.n_lags > 0 and self.config_ar.reg_lambda is not None: 

482 reg_ar = utils.reg_func_ar(self.model.ar_weights) 

483 reg_loss += self.config_ar.reg_lambda * reg_ar 

484 

485 # Regularize trend to be smoother/sparse 

486 l_trend = self.config_trend.trend_reg 

487 if self.config_trend.n_changepoints > 0 and l_trend is not None and l_trend > 0: 

488 reg_trend = utils.reg_func_trend( 

489 weights=self.model.get_trend_deltas, 

490 threshold=self.config_train.trend_reg_threshold, 

491 ) 

492 reg_loss += l_trend * reg_trend 

493 

494 # Regularize seasonality: sparsify fourier term coefficients 

495 l_season = self.config_train.reg_lambda_season 

496 if self.model.season_dims is not None and l_season is not None and l_season > 0: 

497 for name in self.model.season_params.keys(): 

498 reg_season = utils.reg_func_season(self.model.season_params[name]) 

499 reg_loss += l_season * reg_season 

500 

501 # Regularize events: sparsify events features coefficients 

502 if self.events_config is not None or self.country_holidays_config is not None: 

503 reg_events_loss = utils.reg_func_events(self.events_config, self.country_holidays_config, self.model) 

504 reg_loss += reg_events_loss 

505 

506 # Regularize regressors: sparsify regressor features coefficients 

507 if self.regressors_config is not None: 

508 reg_regressor_loss = utils.reg_func_regressors(self.regressors_config, self.model) 

509 reg_loss += reg_regressor_loss 

510 

511 reg_loss = delay_weight * reg_loss 

512 loss = loss + reg_loss 

513 return loss, reg_loss

515 def _evaluate_epoch(self, loader, val_metrics): 

516 """Evaluates model performance. 

517 

518 Args: 

519 loader (torch DataLoader): instantiated Validation Dataloader (with TimeDataset) 

520 val_metrics (MetricsCollection): validation metrics to be computed. 

521 Returns:

532 def _train(self, df, df_val=None, progress_bar=True, plot_live_loss=False): 

533 """Execute model training procedure for a configured number of epochs. 

534 

535 Args: 

536 df (pd.DataFrame): containing column 'ds', 'y' with training data 

537 df_val (pd.DataFrame): containing column 'ds', 'y' with validation data 

538 progress_bar (bool): display updating progress bar 

539 plot_live_loss (bool): plot live training loss, 

540 requires [live] install or livelossplot package installed. 

541 Returns: 

542 df with metrics 

543 """ 

544 if plot_live_loss: 

545 try: 

546 from livelossplot import PlotLosses 

547 except: 

548 plot_live_loss = False 

549 log.warning( 

550 "To plot live loss, please install neuralprophet[live]." 

551 "Using pip: 'pip install neuralprophet[live]'" 

552 "Or install the missing package manually: 'pip install livelossplot'", 

553 exc_info=True, 

554 ) 

555 

556 loader = self._init_train_loader(df) 

557 val = df_val is not None 

558 ## Metrics 

559 if self.highlight_forecast_step_n is not None: 

560 self.metrics.add_specific_target(target_pos=self.highlight_forecast_step_n - 1) 

561 if not self.normalize == "off": 

562 self.metrics.set_shift_scale((self.data_params["y"].shift, self.data_params["y"].scale)) 

563 if val: 

564 val_loader = self._init_val_loader(df_val) 

565 val_metrics = metrics.MetricsCollection([m.new() for m in self.metrics.batch_metrics]) 

566 

567 ## Run 

568 start = time.time() 

569 if progress_bar: 

570 training_loop = tqdm( 

571 range(self.config_train.epochs), total=self.config_train.epochs, leave=log.getEffectiveLevel() <= 20 

572 ) 

573 else: 

574 training_loop = range(self.config_train.epochs) 

575 if plot_live_loss: 

576 live_out = ["MatplotlibPlot"] 

577 if not progress_bar: 

578 live_out.append("ExtremaPrinter") 

579 live_loss = PlotLosses(outputs=live_out) 

580 for e in training_loop: 

581 metrics_live = {} 

582 self.metrics.reset() 

583 if val: 

584 val_metrics.reset() 

585 epoch_metrics = self._train_epoch(e, loader) 

586 metrics_live["{}".format(list(epoch_metrics)[0])] = epoch_metrics[list(epoch_metrics)[0]] 

587 if val: 

588 val_epoch_metrics = self._evaluate_epoch(val_loader, val_metrics) 

589 metrics_live["val_{}".format(list(val_epoch_metrics)[0])] = val_epoch_metrics[ 

590 list(val_epoch_metrics)[0] 

591 ] 

592 print_val_epoch_metrics = {k + "_val": v for k, v in val_epoch_metrics.items()} 

593 else: 

594 val_epoch_metrics = None 

595 print_val_epoch_metrics = OrderedDict() 

596 if progress_bar: 

597 training_loop.set_description(f"Epoch[{(e+1)}/{self.config_train.epochs}]") 

598 training_loop.set_postfix(ordered_dict=epoch_metrics, **print_val_epoch_metrics) 

599 else: 

600 metrics_string = utils.print_epoch_metrics(epoch_metrics, e=e, val_metrics=val_epoch_metrics) 

601 if e == 0: 

602 log.info(metrics_string.splitlines()[0]) 

603 log.info(metrics_string.splitlines()[1]) 

604 else: 

605 log.info(metrics_string.splitlines()[1]) 

606 if plot_live_loss: 

607 live_loss.update(metrics_live) 

608 if plot_live_loss and (e % (1 + self.config_train.epochs // 10) == 0 or e + 1 == self.config_train.epochs): 

609 live_loss.send() 

610 

611 ## Metrics 

612 log.debug("Train Time: {:8.3f}".format(time.time() - start)) 

613 log.debug("Total Batches: {}".format(self.metrics.total_updates)) 

614 metrics_df = self.metrics.get_stored_as_df() 

615 if val: 

616 metrics_df_val = val_metrics.get_stored_as_df() 

617 for col in metrics_df_val.columns: 

618 metrics_df["{}_val".format(col)] = metrics_df_val[col] 

619 return metrics_df

621 def _eval_true_ar(self): 

622 assert self.n_lags > 0 

623 if self.highlight_forecast_step_n is None: 

624 if self.n_lags > 1: 

625 raise ValueError("Please define forecast_lag for sTPE computation") 

626 forecast_pos = 1 

627 else: 

628 forecast_pos = self.highlight_forecast_step_n 

629 weights = self.model.ar_weights.detach().numpy() 

630 weights = weights[forecast_pos - 1, :][::-1] 

631 sTPE = utils.symmetric_total_percentage_error(self.true_ar_weights, weights) 

632 log.info("AR parameters: ", self.true_ar_weights, "\n", "Model weights: ", weights) 

633 return sTPE

768 def make_future_dataframe(self, df, events_df=None, regressors_df=None, periods=None, n_historic_predictions=0): 

769 df = df.copy(deep=True) 

770 if events_df is not None: 

771 events_df = events_df.copy(deep=True).reset_index(drop=True) 

772 if regressors_df is not None: 

773 regressors_df = regressors_df.copy(deep=True).reset_index(drop=True) 

774 n_lags = 0 if self.n_lags is None else self.n_lags 

775 if periods is None: 

776 periods = 1 if n_lags == 0 else self.n_forecasts 

777 else: 

778 assert periods >= 0 

779 

780 if isinstance(n_historic_predictions, bool): 

781 if n_historic_predictions: 

782 n_historic_predictions = len(df) - n_lags 

783 else: 

784 n_historic_predictions = 0 

785 elif not isinstance(n_historic_predictions, int):

787 n_historic_predictions = 0 

788

790 raise ValueError("Set either history or future to contain more than zero values.")

795 raise ValueError("Future values of all user specified regressors not provided") 

796 else: 

797 for regressor in self.regressors_config.keys():

801 last_date = pd.to_datetime(df["ds"].copy(deep=True)).sort_values().max() 

802 

803 if len(df) < n_lags: 

804 raise ValueError("Insufficient data for a prediction") 

805 elif len(df) < n_lags + n_historic_predictions: 

806 log.warning( 

807 "Insufficient data for {} historic forecasts, reduced to {}.".format(

812 if (n_historic_predictions + n_lags) == 0: 

813 df = pd.DataFrame(columns=df.columns) 

814 else: 

815 df = df[-(n_lags + n_historic_predictions) :] 

816 

817 if len(df) > 0: 

818 if len(df.columns) == 1 and "ds" in df: 

819 assert n_lags == 0 

820 df = df_utils.check_dataframe(df, check_y=False) 

821 else: 

822 df = df_utils.check_dataframe( 

823 df, check_y=n_lags > 0, covariates=self.config_covar, events=self.events_config 

824 ) 

825 df = self._handle_missing_data(df, freq=self.data_freq, predicting=True) 

826 df = df_utils.normalize(df, self.data_params) 

827 

828 # future data 

829 # check for external events known in future 

830 if self.events_config is not None and periods > 0 and events_df is None: 

831 log.warning( 

832 "Future values not supplied for user specified events. " 

833 "All events being treated as not occurring in future" 

834 ) 

835 

836 if n_lags > 0: 

837 if periods > 0 and periods != self.n_forecasts: 

838 periods = self.n_forecasts 

839 log.warning( 

840 "Number of forecast steps is defined by n_forecasts. " "Adjusted to {}.".format(self.n_forecasts) 

841 ) 

842 

843 if periods > 0: 

844 future_df = df_utils.make_future_df( 

845 df_columns=df.columns, 

846 last_date=last_date, 

847 periods=periods, 

848 freq=self.data_freq, 

849 events_config=self.events_config, 

850 events_df=events_df, 

851 regressor_config=self.regressors_config, 

852 regressors_df=regressors_df, 

853 ) 

854 future_df = df_utils.normalize(future_df, self.data_params) 

855 if len(df) > 0: 

856 df = df.append(future_df) 

857 else: 

858 df = future_df 

859 df.reset_index(drop=True, inplace=True) 

860 return df

890 def predict(self, df): 

891 """Runs the model to make predictions. 

892 

893 and compute stats (MSE, MAE) 

894 Args: 

895 df (pandas DataFrame): Dataframe with columns 'ds' datestamps, 'y' time series values and 

896 other external variables 

897 

898 Returns: 

899 df_forecast (pandas DataFrame): columns 'ds', 'y', 'trend' and ['yhat<i>'] 

900 """ 

901 # TODO: Implement data sanity checks? 

902 if self.fitted is False: 

903 log.warning("Model has not been fitted. Predictions will be random.") 

904 dataset = self._create_dataset(df, predict_mode=True) 

905 loader = DataLoader(dataset, batch_size=min(1024, len(df)), shuffle=False, drop_last=False) 

906 

907 predicted_vectors = list() 

908 component_vectors = None 

909 with torch.no_grad(): 

910 self.model.eval() 

911 for inputs, _ in loader: 

912 predicted = self.model.forward(inputs) 

913 predicted_vectors.append(predicted.detach().numpy()) 

914 components = self.model.compute_components(inputs) 

915 if component_vectors is None: 

916 component_vectors = {name: [value.detach().numpy()] for name, value in components.items()} 

917 else: 

918 for name, value in components.items(): 

919 component_vectors[name].append(value.detach().numpy()) 

920 components = {name: np.concatenate(value) for name, value in component_vectors.items()} 

921 predicted = np.concatenate(predicted_vectors) 

922 

923 scale_y, shift_y = self.data_params["y"].scale, self.data_params["y"].shift 

924 predicted = predicted * scale_y + shift_y 

925 for name, value in components.items(): 

926 if "multiplicative" in name: 

927 continue 

928 elif "event_" in name: 

929 event_name = name.split("_")[1] 

930 if self.events_config is not None and event_name in self.events_config: 

931 if self.events_config[event_name].mode == "multiplicative": 

932 continue 

933 elif self.country_holidays_config is not None and event_name in self.country_holidays_config: 

934 if self.country_holidays_config[event_name].mode == "multiplicative": 

935 continue 

936 elif "season" in name and self.season_config.mode == "multiplicative": 

937 continue 

938 # scale additive components 

939 components[name] = value * scale_y 

940 if "trend" in name: 

941 components[name] += shift_y 

942 

943 cols = ["ds", "y"] # cols to keep from df 

944 df_forecast = pd.concat((df[cols],), axis=1) 

945 

946 # create a line for each forecast_lag 

947 # 'yhat<i>' is the forecast for 'y' at 'ds' from i steps ago. 

948 for i in range(self.n_forecasts): 

949 forecast_lag = i + 1 

950 forecast = predicted[:, forecast_lag - 1] 

951 pad_before = self.n_lags + forecast_lag - 1 

952 pad_after = self.n_forecasts - forecast_lag 

953 yhat = np.concatenate(([None] * pad_before, forecast, [None] * pad_after)) 

954 df_forecast["yhat{}".format(i + 1)] = yhat 

955 df_forecast["residual{}".format(i + 1)] = yhat - df_forecast["y"] 

956 

957 lagged_components = [ 

958 "ar", 

959 ] 

960 if self.config_covar is not None: 

961 for name in self.config_covar.keys(): 

962 lagged_components.append("lagged_regressor_{}".format(name)) 

963 for comp in lagged_components: 

964 if comp in components: 

965 for i in range(self.n_forecasts): 

966 forecast_lag = i + 1 

967 forecast = components[comp][:, forecast_lag - 1] 

968 pad_before = self.n_lags + forecast_lag - 1 

969 pad_after = self.n_forecasts - forecast_lag 

970 yhat = np.concatenate(([None] * pad_before, forecast, [None] * pad_after)) 

971 df_forecast["{}{}".format(comp, i + 1)] = yhat 

972 

973 # only for non-lagged components 

974 for comp in components: 

975 if comp not in lagged_components: 

976 forecast_0 = components[comp][0, :] 

977 forecast_rest = components[comp][1:, self.n_forecasts - 1] 

978 yhat = np.concatenate(([None] * self.n_lags, forecast_0, forecast_rest)) 

979 df_forecast[comp] = yhat 

980 return df_forecast
