17def init_data_params(df, normalize, covariates_config=None, regressor_config=None, events_config=None): 

18 """Initialize data scaling values. 

19 

20 Note: We do a z normalization on the target series 'y', 

21 unlike OG Prophet, which does shift by min and scale by max. 

22 Args: 

23 df (pd.DataFrame): Time series to compute normalization parameters from. 

24 normalize (str): Type of normalization to apply to the time series. 

25 options: ['soft', 'off', 'minmax, 'standardize'] 

26 default: 'soft' scales minimum to 0.1 and the 90th quantile to 0.9 

27 covariates_config (OrderedDict): extra regressors with sub_parameters 

28 normalize (bool) 

29 regressor_config (OrderedDict): extra regressors (with known future values) 

30 with sub_parameters normalize (bool) 

31 events_config (OrderedDict): user specified events configs 

32 

33 Returns: 

34 data_params (OrderedDict): scaling values 

35 with ShiftScale entries containing 'shift' and 'scale' parameters 

36 """ 

37 data_params = OrderedDict({}) 

38 

39 if df["ds"].dtype == np.int64: 

40 df.loc[:, "ds"] = df.loc[:, "ds"].astype(str) 

41 df.loc[:, "ds"] = pd.to_datetime(df.loc[:, "ds"]) 

42 data_params["ds"] = ShiftScale( 

43 shift=df["ds"].min(), 

44 scale=df["ds"].max() - df["ds"].min(), 

45 ) 

46 

47 if "y" in df: 

48 data_params["y"] = get_normalization_params( 

49 array=df["y"].values, 

50 norm_type=normalize, 

51 ) 

52 

53 if covariates_config is not None: 

54 for covar in covariates_config.keys(): 

55 if covar not in df.columns: 

56 raise ValueError("Covariate {} not found in DataFrame.".format(covar)) 

57 data_params[covar] = get_normalization_params( 

58 array=df[covar].values, 

59 norm_type=covariates_config[covar].normalize, 

60 ) 

61 

62 if regressor_config is not None: 

63 for reg in regressor_config.keys(): 

64 if reg not in df.columns: 

65 raise ValueError("Regressor {} not found in DataFrame.".format(reg)) 

66 data_params[reg] = get_normalization_params( 

67 array=df[reg].values, 

68 norm_type=regressor_config[reg].normalize, 

69 ) 

70 if events_config is not None: 

71 for event in events_config.keys(): 

72 if event not in df.columns: 

73 raise ValueError("Event {} not found in DataFrame.".format(event)) 

74 data_params[event] = ShiftScale() 

75 log.debug("Data Parameters (shift, scale): {}".format([(k, (v.shift, v.scale)) for k, v in data_params.items()])) 

76 return data_params

79def auto_normalization_setting(array): 

80 if len(np.unique(array)) < 2: 

81 log.error("encountered variable with one unique value") 

82 raise ValueError 

83 # elif set(series.unique()) in ({True, False}, {1, 0}, {1.0, 0.0}, {-1, 1}, {-1.0, 1.0}): 

84 elif len(np.unique(array)) == 2: 

85 return "minmax" # Don't standardize binary variables. 

86 else: 

87 return "soft" # default setting

90def get_normalization_params(array, norm_type): 

91 if norm_type == "auto": 

92 norm_type = auto_normalization_setting(array) 

93 shift = 0.0 

94 scale = 1.0 

95 if norm_type == "soft": 

96 lowest = np.min(array) 

97 q95 = np.quantile(array, 0.95, interpolation="higher") 

98 width = q95 - lowest 

99 if math.isclose(width, 0): 

100 width = np.max(array) - lowest 

101 shift = lowest 

102 scale = width 

103 elif norm_type == "soft1": 

104 lowest = np.min(array) 

105 q90 = np.quantile(array, 0.9, interpolation="higher") 

106 width = q90 - lowest 

107 if math.isclose(width, 0): 

108 width = (np.max(array) - lowest) / 1.25 

109 shift = lowest - 0.125 * width 

110 scale = 1.25 * width 

111 elif norm_type == "minmax": 

112 shift = np.min(array) 

113 scale = np.max(array) - shift 

114 elif norm_type == "standardize": 

115 shift = np.mean(array) 

116 scale = np.std(array) 

117 elif norm_type != "off": 

118 log.error("Normalization {} not defined.".format(norm_type)) 

119 return ShiftScale(shift, scale)

122def normalize(df, data_params): 

123 """Apply data scales. 

124 

125 Applies data scaling factors to df using data_params. 

126 

127 Args: 

128 df (pd.DataFrame): with columns 'ds', 'y', (and potentially more regressors) 

129 data_params (OrderedDict): scaling values,as returned by init_data_params 

130 with ShiftScale entries containing 'shift' and 'scale' parameters 

131 Returns: 

132 df: pd.DataFrame, normalized 

133 """ 

134 for name in df.columns: 

135 if name not in data_params.keys(): 

136 raise ValueError("Unexpected column {} in data".format(name)) 

137 new_name = name 

138 if name == "ds": 

139 new_name = "t" 

140 if name == "y": 

141 new_name = "y_scaled" 

142 df[new_name] = df[name].sub(data_params[name].shift).div(data_params[name].scale) 

143 return df

146def check_dataframe(df, check_y=True, covariates=None, regressors=None, events=None): 

147 """Performs basic data sanity checks and ordering 

148 

149 Prepare dataframe for fitting or predicting. 

150 Args: 

151 df (pd.DataFrame): with columns ds 

152 check_y (bool): if df must have series values 

153 set to True if training or predicting with autoregression 

154 covariates (list or dict): covariate column names 

155 regressors (list or dict): regressor column names 

156 events (list or dict): event column names 

157 

158 Returns: 

159 pd.DataFrame 

160 """ 

161 if df.shape[0] == 0: 

162 raise ValueError("Dataframe has no rows.") 

163 

164 if "ds" not in df: 

165 raise ValueError('Dataframe must have columns "ds" with the dates.') 

166 if df.loc[:, "ds"].isnull().any(): 

167 raise ValueError("Found NaN in column ds.") 

168 if df["ds"].dtype == np.int64: 

169 df.loc[:, "ds"] = df.loc[:, "ds"].astype(str) 

170 if not np.issubdtype(df["ds"].dtype, np.datetime64): 

171 df.loc[:, "ds"] = pd.to_datetime(df.loc[:, "ds"]) 

172 if df["ds"].dt.tz is not None: 

173 raise ValueError("Column ds has timezone specified, which is not supported. Remove timezone.") 

174 

175 columns = [] 

176 if check_y: 

177 columns.append("y") 

178 if covariates is not None: 

179 if type(covariates) is list: 

180 columns.extend(covariates) 

181 else: # treat as dict 

182 columns.extend(covariates.keys()) 

183 if regressors is not None: 

184 if type(regressors) is list: 

185 columns.extend(regressors) 

186 else: # treat as dict 

187 columns.extend(regressors.keys()) 

188 if events is not None: 

189 if type(events) is list: 

190 columns.extend(events) 

191 else: # treat as dict 

192 columns.extend(events.keys()) 

193 for name in columns: 

194 if name not in df: 

195 raise ValueError("Column {name!r} missing from dataframe".format(name=name)) 

196 if df.loc[df.loc[:, name].notnull()].shape[0] < 1: 

197 raise ValueError("Dataframe column {name!r} only has NaN rows.".format(name=name)) 

198 if not np.issubdtype(df[name].dtype, np.number): 

199 df.loc[:, name] = pd.to_numeric(df.loc[:, name]) 

200 if np.isinf(df.loc[:, name].values).any(): 

201 df.loc[:, name] = df[name].replace([np.inf, -np.inf], np.nan) 

202 if df.loc[df.loc[:, name].notnull()].shape[0] < 1: 

203 raise ValueError("Dataframe column {name!r} only has NaN rows.".format(name=name)) 

204 

205 if df.index.name == "ds": 

206 df.index.name = None 

207 df = df.sort_values("ds") 

208 df = df.reset_index(drop=True) 

209 return df

212def crossvalidation_split_df(df, n_lags, n_forecasts, k, fold_pct, fold_overlap_pct=0.0): 

213 """Splits data in k folds for crossvalidation. 

214 

215 Args: 

216 df (pd.DataFrame): data 

217 n_lags (int): identical to NeuralProhet 

218 n_forecasts (int): identical to NeuralProhet 

219 k: number of CV folds 

220 fold_pct: percentage of overall samples to be in each fold 

221 fold_overlap_pct: percentage of overlap between the validation folds. 

222 default: 0.0 

223 

224 Returns: 

225 list of k tuples [(df_train, df_val), ...] where: 

226 df_train (pd.DataFrame): training data 

227 df_val (pd.DataFrame): validation data 

228 """ 

229 if n_lags == 0: 

230 assert n_forecasts == 1 

231 total_samples = len(df) - n_lags + 2 - (2 * n_forecasts) 

232 samples_fold = max(1, int(fold_pct * total_samples)) 

233 samples_overlap = int(fold_overlap_pct * samples_fold) 

234 assert samples_overlap < samples_fold 

235 min_train = total_samples - samples_fold - (k - 1) * (samples_fold - samples_overlap) 

236 assert min_train >= samples_fold 

237 folds = [] 

238 df_fold = df.copy(deep=True) 

239 for i in range(k, 0, -1): 

240 df_train, df_val = split_df(df_fold, n_lags, n_forecasts, valid_p=samples_fold, inputs_overbleed=True) 

241 folds.append((df_train, df_val)) 

242 split_idx = len(df_fold) - samples_fold + samples_overlap 

243 df_fold = df_fold.iloc[:split_idx].reset_index(drop=True) 

244 folds = folds[::-1] 

245 return folds

248def split_df(df, n_lags, n_forecasts, valid_p=0.2, inputs_overbleed=True): 

249 """Splits timeseries df into train and validation sets. 

250 

251 Prevents overbleed of targets. Overbleed of inputs can be configured. 

252 

253 Args: 

254 df (pd.DataFrame): data 

255 n_lags (int): identical to NeuralProhet 

256 n_forecasts (int): identical to NeuralProhet 

257 valid_p (float, int): fraction (0,1) of data to use for holdout validation set, 

258 or number of validation samples >1 

259 inputs_overbleed (bool): Whether to allow last training targets to be first validation inputs (never targets) 

260 

261 Returns: 

262 df_train (pd.DataFrame): training data 

263 df_val (pd.DataFrame): validation data 

264 """ 

265 n_samples = len(df) - n_lags + 2 - (2 * n_forecasts) 

266 n_samples = n_samples if inputs_overbleed else n_samples - n_lags 

267 if 0.0 < valid_p < 1.0: 

268 n_valid = max(1, int(n_samples * valid_p)) 

269 else: 

270 assert valid_p >= 1 

271 assert type(valid_p) == int 

272 n_valid = valid_p 

273 n_train = n_samples - n_valid 

274 assert n_train >= 1 

275 

276 split_idx_train = n_train + n_lags + n_forecasts - 1 

277 split_idx_val = split_idx_train - n_lags if inputs_overbleed else split_idx_train 

278 df_train = df.copy(deep=True).iloc[:split_idx_train].reset_index(drop=True) 

279 df_val = df.copy(deep=True).iloc[split_idx_val:].reset_index(drop=True) 

280 log.debug("{} n_train, {} n_eval".format(n_train, n_samples - n_train)) 

281 return df_train, df_val

365def fill_linear_then_rolling_avg(series, limit_linear, rolling): 

366 """Adds missing dates, fills missing values with linear imputation or trend. 

367 

368 Args: 

369 series (pd.Series): series with nan to be filled in. 

370 limit_linear (int): maximum number of missing values to impute. 

371 Note: because imputation is done in both directions, this value is effectively doubled. 

372 rolling (int): maximal number of missing values to impute. 

373 Note: window width is rolling + 2*limit_linear 

374 

375 Returns: 

376 filled df 

377 """ 

378 # impute small gaps linearly: 

379 series = series.interpolate(method="linear", limit=limit_linear, limit_direction="both") 

380 # fill remaining gaps with rolling avg 

381 is_na = pd.isna(series) 

382 rolling_avg = series.rolling(rolling + 2 * limit_linear, min_periods=2 * limit_linear, center=True).mean() 

383 series.loc[is_na] = rolling_avg[is_na] 

384 remaining_na = sum(series.isnull()) 

385 return series, remaining_na
