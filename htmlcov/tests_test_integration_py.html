535 def check_cv(df, freq, n_lags, n_forecasts, k, fold_pct, fold_overlap_pct): 

536 m = NeuralProphet( 

537 n_lags=n_lags, 

538 n_forecasts=n_forecasts, 

539 ) 

540 folds = m.crossvalidation_split_df(df, freq=freq, k=k, fold_pct=fold_pct, fold_overlap_pct=fold_overlap_pct) 

541 total_samples = len(df) - m.n_lags + 2 - (2 * m.n_forecasts) 

542 per_fold = int(fold_pct * total_samples) 

543 not_overlap = per_fold - int(fold_overlap_pct * per_fold) 

544 assert all([per_fold == len(val) - m.n_lags + 1 - m.n_forecasts for (train, val) in folds]) 

545 assert all( 

546 [ 

547 total_samples - per_fold - (k - i - 1) * not_overlap == len(train) - m.n_lags + 1 - m.n_forecasts 

548 for i, (train, val) in enumerate(folds) 

549 ] 

550 ) 

551 

552 check_simple(pd.DataFrame({"ds": pd.date_range(start="2017-01-01", periods=100), "y": np.arange(100)}))
