17def reg_func_ar(weights): 

18 """Regularization of coefficients based on AR-Net paper 

19 

20 Args: 

21 weights (torch tensor): Model weights to be regularized towards zero 

22 

23 Returns: 

24 regularization loss, scalar 

25 

26 """ 

27 # reg = torch.div(2.0, 1.0 + torch.exp(-2 * (1e-9 + torch.abs(weights)).pow(1 / 2.0))) - 1.0 

28 # reg = torch.abs(weights) 

29 reg = torch.log(0.1 + torch.abs(weights)) - torch.log(0.1 * torch.ones(1)) 

30 reg = torch.mean(reg).squeeze() 

31 return reg

46def reg_func_trend(weights, threshold=None): 

47 """Regularization of weights to induce sparcity 

48 

49 Args: 

50 weights (torch tensor): Model weights to be regularized towards zero 

51 threshold (float): value below which not to regularize weights 

52 

53 Returns: 

54 regularization loss, scalar 

55 """ 

56 abs_weights = torch.abs(weights) 

57 if threshold is not None and not math.isclose(threshold, 0): 

58 abs_weights = torch.clamp(abs_weights - threshold, min=0.0) 

59 reg = torch.sum(abs_weights).squeeze() 

60 return reg

67def reg_func_events(events_config, country_holidays_config, model): 

68 """ 

69 Regularization of events coefficients to induce sparcity 

70 

71 Args: 

72 events_config (OrderedDict): Configurations (upper, lower windows, regularization) for user specified events 

73 country_holidays_config (OrderedDict): Configurations (holiday_names, upper, lower windows, regularization) 

74 for country specific holidays 

75 model (TimeNet): The TimeNet model object 

76 

77 Returns: 

78 regularization loss, scalar 

79 """ 

80 reg_events_loss = 0.0 

81 if events_config is not None: 

82 for event, configs in events_config.items(): 

83 reg_lambda = configs["trend_reg"] 

84 if reg_lambda is not None: 

85 weights = model.get_event_weights(event) 

86 for offset in weights.keys(): 

87 reg_events_loss += reg_lambda * reg_func_abs(weights[offset]) 

88 

89 if country_holidays_config is not None: 

90 reg_lambda = country_holidays_config["trend_reg"] 

91 if reg_lambda is not None: 

92 for holiday in country_holidays_config["holiday_names"]: 

93 weights = model.get_event_weights(holiday) 

94 for offset in weights.keys(): 

95 reg_events_loss += reg_lambda * reg_func_abs(weights[offset]) 

96 

97 return reg_events_loss

100def reg_func_regressors(regressors_config, model): 

101 """ 

102 Regularization of regressors coefficients to induce sparcity 

103 Args: 

104 regressors_config (OrderedDict): Configurations for user specified regressors 

105 model (TimeNet): The TimeNet model object 

106 Returns: 

107 regularization loss, scalar 

108 """ 

109 reg_regressor_loss = 0.0

119def symmetric_total_percentage_error(values, estimates): 

120 """Compute STPE 

121 

122 Args: 

123 values (np.array): 

124 estimates (np.array): 

125 

126 Returns: 

127 scalar (float) 

128 """ 

129 sum_abs_diff = np.sum(np.abs(estimates - values)) 

130 sum_abs = np.sum(np.abs(estimates) + np.abs(values)) 

131 return 100 * sum_abs_diff / (10e-9 + sum_abs)

182def events_config_to_model_dims(events_config, country_holidays_config): 

183 """ 

184 Convert the NeuralProphet user specified events configurations along with country specific 

185 holidays to input dims for TimeNet model. 

186 Args: 

187 events_config (OrderedDict): Configurations (upper, lower windows, regularization) for user specified events 

188 country_holidays_config (OrderedDict): Configurations (holiday_names, upper, lower windows, regularization) 

189 for country specific holidays 

190 

191 Returns: 

192 events_dims (OrderedDict): A dictionary with keys corresponding to individual holidays and values in an AttrDict 

193 with configs such as the mode, list of event delims of the event corresponding to the offsets, 

194 and the indices in the input dataframe corresponding to each event. 

195 """ 

196 if events_config is None and country_holidays_config is None: 

197 return None 

198 additive_events_dims = pd.DataFrame(columns=["event", "event_delim"]) 

199 multiplicative_events_dims = pd.DataFrame(columns=["event", "event_delim"]) 

200 

201 if events_config is not None: 

202 for event, configs in events_config.items(): 

203 mode = configs["mode"] 

204 for offset in range(configs.lower_window, configs.upper_window + 1): 

205 event_delim = create_event_names_for_offsets(event, offset) 

206 if mode == "additive": 

207 additive_events_dims = additive_events_dims.append( 

208 {"event": event, "event_delim": event_delim}, ignore_index=True 

209 ) 

210 else: 

211 multiplicative_events_dims = multiplicative_events_dims.append( 

212 {"event": event, "event_delim": event_delim}, ignore_index=True 

213 ) 

214 

215 if country_holidays_config is not None: 

216 lower_window = country_holidays_config["lower_window"] 

217 upper_window = country_holidays_config["upper_window"] 

218 mode = country_holidays_config["mode"] 

219 for country_holiday in country_holidays_config["holiday_names"]: 

220 for offset in range(lower_window, upper_window + 1): 

221 holiday_delim = create_event_names_for_offsets(country_holiday, offset) 

222 if mode == "additive": 

223 additive_events_dims = additive_events_dims.append( 

224 {"event": country_holiday, "event_delim": holiday_delim}, ignore_index=True 

225 ) 

226 else: 

227 multiplicative_events_dims = multiplicative_events_dims.append( 

228 {"event": country_holiday, "event_delim": holiday_delim}, ignore_index=True 

229 ) 

230 

231 # sort based on event_delim 

232 event_dims = pd.DataFrame() 

233 if not additive_events_dims.empty: 

234 additive_events_dims = additive_events_dims.sort_values(by="event_delim").reset_index(drop=True) 

235 additive_events_dims["mode"] = "additive" 

236 event_dims = additive_events_dims 

237 

238 if not multiplicative_events_dims.empty: 

239 multiplicative_events_dims = multiplicative_events_dims.sort_values(by="event_delim").reset_index(drop=True) 

240 multiplicative_events_dims["mode"] = "multiplicative" 

241 event_dims = event_dims.append(multiplicative_events_dims) 

242 

243 event_dims_dic = OrderedDict({}) 

244 # convert to dict format 

245 for event, row in event_dims.groupby("event"): 

246 event_dims_dic[event] = AttrDict( 

247 {"mode": row["mode"].iloc[0], "event_delim": list(row["event_delim"]), "event_indices": list(row.index)} 

248 ) 

249 return event_dims_dic

312def set_auto_seasonalities(dates, season_config): 

313 """Set seasonalities that were left on auto or set by user. 

314 

315 Turns on yearly seasonality if there is >=2 years of history. 

316 Turns on weekly seasonality if there is >=2 weeks of history, and the 

317 spacing between dates in the history is <7 days. 

318 Turns on daily seasonality if there is >=2 days of history, and the 

319 spacing between dates in the history is <1 day. 

320 

321 Args: 

322 dates (pd.Series): datestamps 

323 season_config (configure.AllSeason): NeuralProphet seasonal model configuration, as after __init__ 

324 Returns: 

325 season_config (configure.AllSeason): processed NeuralProphet seasonal model configuration 

326 

327 """ 

328 log.debug("seasonality config received: {}".format(season_config)) 

329 first = dates.min() 

330 last = dates.max() 

331 dt = dates.diff() 

332 min_dt = dt.iloc[dt.values.nonzero()[0]].min() 

333 auto_disable = { 

334 "yearly": last - first < pd.Timedelta(days=730), 

335 "weekly": ((last - first < pd.Timedelta(weeks=2)) or (min_dt >= pd.Timedelta(weeks=1))), 

336 "daily": ((last - first < pd.Timedelta(days=2)) or (min_dt >= pd.Timedelta(days=1))), 

337 } 

338 for name, period in season_config.periods.items(): 

339 arg = period.arg 

340 default_resolution = period.resolution 

341 if arg == "custom": 

342 continue 

343 elif arg == "auto": 

344 resolution = 0 

345 if auto_disable[name]: 

346 log.info( 

347 "Disabling {name} seasonality. Run NeuralProphet with " 

348 "{name}_seasonality=True to override this.".format(name=name) 

349 ) 

350 else: 

351 resolution = default_resolution 

352 elif arg is True: 

353 resolution = default_resolution 

354 elif arg is False: 

355 resolution = 0 

356 else: 

357 resolution = int(arg) 

358 season_config.periods[name].resolution = resolution 

359 

360 new_periods = OrderedDict({}) 

361 for name, period in season_config.periods.items(): 

362 if period.resolution > 0: 

363 new_periods[name] = period 

364 season_config.periods = new_periods 

365 season_config = season_config if len(season_config.periods) > 0 else None 

366 log.debug("seasonality config: {}".format(season_config)) 

367 return season_config

384def fcst_df_to_last_forecast(fcst, n_last=1): 

385 """Converts from line-per-lag to line-per-forecast. 

386 

387 Args: 

388 fcst (pd.DataFrame): forecast df 

389 n_last (int): number of last forecasts to include 

390 

391 Returns: 

392 df where yhat1 is last forecast, yhat2 second to last etc 

393 """ 

394 

395 cols = ["ds", "y"] # cols to keep from df 

396 df = pd.concat((fcst[cols],), axis=1) 

397 df.reset_index(drop=True, inplace=True) 

398 

399 yhat_col_names = [col_name for col_name in fcst.columns if "yhat" in col_name] 

400 n_forecast_steps = len(yhat_col_names) 

401 yhats = pd.concat((fcst[yhat_col_names],), axis=1) 

402 cols = list(range(n_forecast_steps)) 

403 for i in range(n_last - 1, -1, -1): 

404 forecast_name = "yhat{}".format(i + 1) 

405 df[forecast_name] = None 

406 rows = len(df) + np.arange(-n_forecast_steps - i, -i, 1) 

407 last = yhats.values[rows, cols] 

408 df.loc[rows, forecast_name] = last 

409 return df
